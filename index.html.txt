const chatMessages=document.getElementById('chatMessages');
let chatHistory=[];
async function sendChat(){
  const input=document.getElementById('chatInput');
  const text=input.value.trim();
  if(!text)return;
  input.value='';
  const um=document.createElement('div');
  um.className='msg user';
  um.innerHTML=text+'<div class="msg-time">'+getTime()+'</div>';
  chatMessages.appendChild(um);
  chatMessages.scrollTop=chatMessages.scrollHeight;
  chatHistory.push({role:'user',content:text});
  const typing=document.createElement('div');
  typing.className='msg bot';
  typing.innerHTML='<span class="typing"><span></span><span></span><span></span></span>';
  chatMessages.appendChild(typing);
  chatMessages.scrollTop=chatMessages.scrollHeight;
  try{
    const r=await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',headers:{'Content-Type':'application/json'},
      body:JSON.stringify({
        model:'claude-sonnet-4-6',max_tokens:1000,
        system:'أنت "رفيق عضُد" — مساعد ذكاء اصطناعي متخصص في دعم الحجاج والمعتمرين من ذوي الإعاقة. تجيب على أسئلة الحج والصحة والتنقل في مكة المكرمة. الحرارة الحالية 42 درجة مئوية. ردودك مختصرة ودافئة بالعربية فقط.',
        messages:chatHistory
      })
    });
    const d=await r.json();
    const reply=d.content?.[0]?.text||'عذراً، حدث خطأ. أعد المحاولة.';
    chatHistory.push({role:'assistant',content:reply});
    typing.innerHTML=reply+'<div class="msg-time">'+getTime()+'</div>';
  }catch(e){
    typing.innerHTML='حدث خطأ في الاتصال.<div class="msg-time">'+getTime()+'</div>';
  }
  chatMessages.scrollTop=chatMessages.scrollHeight;
}
function getTime(){return new Date().toLocaleTimeString('ar',{hour:'2-digit',minute:'2-digit'})}

   ======================================================== */
const MECCA={lat:21.4225,lon:39.8262};
async function loadWeather(lat,lon,label){
  const card=document.getElementById('weatherCard');
  if(!card)return;
  try{
    const url='https://api.open-meteo.com/v1/forecast?latitude='+lat+'&longitude='+lon+
      '&current=temperature_2m,relative_humidity_2m,wind_speed_10m,uv_index&hourly=uv_index&timezone=auto';
    const aqUrl='https://air-quality-api.open-meteo.com/v1/air-quality?latitude='+lat+'&longitude='+lon+'&current=us_aqi';
    const [wRes,aRes]=await Promise.all([fetch(url),fetch(aqUrl).catch(()=>null)]);
    const w=await wRes.json();
    let aqi='—';
    if(aRes){try{const a=await aRes.json();aqi=a.current?.us_aqi??'—';}catch(e){}}
    const t=w.current.temperature_2m, h=w.current.relative_humidity_2m,
          ws=w.current.wind_speed_10m, uv=w.current.uv_index;
    let icon='☀️'; if(t>40)icon='🔥'; else if(uv>8)icon='🌡️'; else if(h>60)icon='🌥️';
    card.innerHTML=
      '<div class="weather-head">'+
        '<div><div class="weather-temp">'+Math.round(t)+'°م</div>'+
        '<div class="weather-loc">'+(label||'موقعك الحالي')+' — تحديث لحظي</div></div>'+
        '<div class="weather-icon">'+icon+'</div>'+
      '</div>'+
      '<div class="weather-grid">'+
        '<div class="weather-stat"><div class="wv">'+Math.round(h)+'%</div><div class="wl">الرطوبة</div></div>'+
        '<div class="weather-stat"><div class="wv">'+Math.round(ws)+' كم/س</div><div class="wl">سرعة الرياح</div></div>'+
        '<div class="weather-stat"><div class="wv">'+(uv!=null?uv.toFixed(1):'—')+'</div><div class="wl">مؤشر UV</div></div>'+
        '<div class="weather-stat"><div class="wv">'+aqi+'</div><div class="wl">جودة الهواء</div></div>'+
      '</div>';
    window.__lastWeather={t,h,ws,uv,aqi};
    updateRiskFromWeather(t,h,uv);
  }catch(e){
    card.innerHTML='<div class="weather-loading">تعذّر تحميل بيانات الطقس حالياً. سيُعاد المحاولة...</div>';
    setTimeout(()=>loadWeather(lat,lon,label),8000);
  }
}
function updateRiskFromWeather(t,h,uv){
  // تحديث لحظي لمستوى خطر الإجهاد الحراري في لوحة التحكم بناءً على الطقس الحقيقي
  const circle=document.querySelector('.risk-num');
  const lbl=document.querySelector('.risk-lbl');
  if(!circle)return;
  let score=Math.min(100,Math.max(0,Math.round((t-25)*3 + (uv||0)*2 + (h<20?10:0))));
  circle.textContent=score+'%';
  if(lbl){
    if(score>70){lbl.textContent='مرتفع';lbl.style.color='var(--danger)';}
    else if(score>40){lbl.textContent='متوسط';lbl.style.color='var(--warn)';}
    else{lbl.textContent='منخفض';lbl.style.color='var(--safe)';}
  }
}

/* ========================================================
   LIVE MAP — Leaflet + OpenStreetMap (free, no API key)
   ======================================================== */
let liveMap,userMarker,routeLine,currentRouteMode='safe';
const HAJJ_SITES=[
  {name:'المسجد الحرام',lat:21.4225,lon:39.8262,type:'landmark'},
  {name:'منى',lat:21.4133,lon:39.8933,type:'landmark'},
  {name:'عرفات',lat:21.3556,lon:39.9842,type:'landmark'},
  {name:'مزدلفة',lat:21.3831,lon:39.9361,type:'landmark'},
  {name:'نقطة إسعاف 1 — المسعى',lat:21.4231,lon:39.8268,type:'medical'},
  {name:'نقطة إسعاف 2 — الجمرات',lat:21.4222,lon:39.8740,type:'medical'},
  {name:'فندق هيلتون مكة',lat:21.4189,lon:39.8255,type:'hotel'}
];
const CONGESTION_ZONES=[
  {name:'المسعى',lat:21.4231,lon:39.8268,level:67},
  {name:'الجمرات',lat:21.4222,lon:39.8740,level:58},
  {name:'الطواف',lat:21.4225,lon:39.8262,level:72}
];
function initLiveMap(){
  if(liveMap)return;
  const el=document.getElementById('liveMap');
  if(!el || typeof L==='undefined')return;
  liveMap=L.map('liveMap',{zoomControl:true}).setView([MECCA.lat,MECCA.lon],15);
  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png',{
    maxZoom:19,attribution:'© OpenStreetMap'
  }).addTo(liveMap);

  HAJJ_SITES.forEach(s=>{
    const color = s.type==='medical' ? '#A17136' : s.type==='hotel' ? '#1D6B6B' : '#3E2A12';
    L.circleMarker([s.lat,s.lon],{radius:8,color:color,fillColor:color,fillOpacity:.85,weight:2})
      .addTo(liveMap).bindPopup('<b>'+s.name+'</b>');
  });

  CONGESTION_ZONES.forEach(z=>{
    const c = z.level>65 ? '#C0392B' : z.level>50 ? '#E67E22' : '#2D7D4E';
    L.circle([z.lat,z.lon],{radius:120,color:c,fillColor:c,fillOpacity:.25,weight:1})
      .addTo(liveMap).bindPopup('<b>'+z.name+'</b><br/>كثافة الازدحام: '+z.level+'%');
  });

  drawRoute(currentRouteMode);
  document.getElementById('mapStatus').textContent='الخريطة جاهزة — اضغط على أي نقطة لعرض التفاصيل';
}
function drawRoute(mode){
  if(!liveMap)return;
  if(routeLine)liveMap.removeLayer(routeLine);
  // مسارات تجريبية مبنية على إحداثيات حقيقية بين المسجد الحرام ومنى، تتغيّر حسب نوع المسار المختار
  const routes={
    safe:[[21.4225,39.8262],[21.4205,39.8420],[21.4160,39.8650],[21.4133,39.8933]],
    shortest:[[21.4225,39.8262],[21.4170,39.8600],[21.4133,39.8933]],
    fastest:[[21.4225,39.8262],[21.4250,39.8500],[21.4190,39.8750],[21.4133,39.8933]],
    wheelchair:[[21.4225,39.8262],[21.4210,39.8400],[21.4185,39.8600],[21.4150,39.8800],[21.4133,39.8933]]
  };
  const colors={safe:'#2D7D4E',shortest:'#1D6B6B',fastest:'#A17136',wheelchair:'#2A8F8F'};
  routeLine=L.polyline(routes[mode],{color:colors[mode],weight:5,opacity:.85,dashArray: mode==='wheelchair'?'8,6':null}).addTo(liveMap);
  liveMap.fitBounds(routeLine.getBounds(),{padding:[30,30]});
}
function setRouteMode(mode,btn){
  currentRouteMode=mode;
  document.querySelectorAll('.map-toolbar button[data-route]').forEach(b=>b.classList.remove('active'));
  if(btn)btn.classList.add('active');
  drawRoute(mode);
  const labels={safe:'المسار الآمن (يتجنب الازدحام)',shortest:'أقصر طريق',fastest:'أسرع طريق',wheelchair:'مسار مخصص للكراسي المتحركة'};
  document.getElementById('mapStatus').textContent='تم عرض: '+labels[mode];
}
function locateUser(){
  const status=document.getElementById('mapStatus');
  if(!navigator.geolocation){status.textContent='المتصفح لا يدعم تحديد الموقع';return;}
  status.textContent='جارٍ تحديد موقعك...';
  navigator.geolocation.getCurrentPosition(pos=>{
    const {latitude:lat,longitude:lon}=pos.coords;
    if(userMarker)liveMap.removeLayer(userMarker);
    userMarker=L.circleMarker([lat,lon],{radius:9,color:'#3E2A12',fillColor:'#A17136',fillOpacity:1,weight:3})
      .addTo(liveMap).bindPopup('موقعك الحالي').openPopup();
    liveMap.setView([lat,lon],16);
    status.textContent='تم تحديد موقعك الحالي بنجاح';
    loadWeather(lat,lon,'موقعك الحالي');
  },()=>{
    status.textContent='تعذّر الوصول للموقع — تأكد من السماح بإذن الموقع';
  },{enableHighAccuracy:true,timeout:8000});
}
